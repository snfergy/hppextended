	give_back_territory_to_GER = {
		potential = {
			or = {
				has_country_flag = "giving_back_territory"
				ai = yes
			}
			exists = GER
			GER = { government_in_exile = no }
			any_owned_province = {
				is_core = GER
				or = {
					THIS = { vassal_of = GER }
					not = { is_core = THIS }
				}
			}
			not = { war_with = GER }
			or = {
				GER = { faction = THIS }
				is_in_any_faction = no
				ai = no
			}
			or = {
				ai = no
				alliance_with = GER			# Or simply allies
				GER = { faction = THIS }	# Same Faction
				and = {
					ideology_group = democracy		# For democrats:
					or = {							# Also democrat and not enemy
						and = {
							relation = { who = GER value = 150 }
							GER = { ideology_group = democracy }
						}
						relation = { who = GER value = 200 }
					}
				}
				and = {
					not = { ideology_group = democracy }		# For non-democrats:
					relation = { who = GER value = 175 }		# Good friends
					GER = { ideology_group = democracy }
				}
			}
		}
		allow = {
			not = {
				any_neighbor_country = {
					war_with = THIS
				}
			}
		}
		effect = {
			any_owned = {
				limit = {
					or = {
						not = { is_core = THIS }
						THIS = { vassal_of = GER }
					}
					is_core = GER
				}
				secede_province = GER
			}
			relation = { who = GER value = 50 }
		}
