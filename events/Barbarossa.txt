country_event = {

	id = 3701

	major = yes
	
	trigger = {
		or = {
			faction = axis
			alliance_with = GER
		}
		any_owned_province = {
			is_capital = yes
			OR = {
				continent = europe
				continent = Scandinavia
				continent = Baltic_States
				continent = Iberia
				continent = Balkans
				continent = Arabia
			}
		}
	}

	is_triggered_only = yes

	title = "Operation Barbarossa"
	desc = "Germany is planning an attack on the Soviet Union, codenamed: Operation Barbarossa. They are asking for our help in that plan."
	picture = "barbarossa"

	option = {
		name = "We stand by the pact!"
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0.0
				any_owned_province = {
					or = {
						and = {
							any_neighbor_province = { owned_by = SOV controlled_by = SOV units_in_province = 1 }
							not = { units_in_province = 1 }		# The borders are undefended!
						}		
						and = {
							any_neighbor_province = { owned_by = SOV controlled_by = SOV units_in_province = 4 }
							not = { units_in_province = 2 }		# We are outnumbered 2:1!!!
						}
						and = {
							any_neighbor_province = { owned_by = SOV controlled_by = SOV units_in_province = 6 }
							not = { units_in_province = 3 }		# We are outnumbered 2:1!!!
						}										# If we have three units in a province,
					}											# that should be enough until reinforcements arrive.
				}
			}
		}
		war = {
			target = SOV
			attacker_goal = { casus_belli = aquire_all_cores }
		}
		GER = { country_event = 3702 }
	}
	option = {
		name = "We can't afford a war right now!"
		ai_chance = {
			factor = 10
		}
		GER = { country_event = 3703 }
	}
}

	id = 3705

	is_triggered_only = yes

	title = "Operation Barbarossa!"
	desc = "The Germans have launched an invasion on the Soviet Union, codenamed Operation Barbarossa."
	picture = "war_declared"
	
	
	trigger = {
		# 1.) Can't be at war with GER
		not = { war_with = GER }
		or = {
			# 2.) One of 2 options: neighbor GER and not AXIS or COM || Democratic AND major TAG
			and = {
				not = { faction = axis }
				not = { faction = comintern }
				any_neighbor_province = { owned_by = GER }
			}
			and = {
				ideology_group = democracy
				or = {
					tag = USA
					tag = ENG
					tag = FRA
					tag = ITA
					tag = SPR
					tag = SPA
					tag = JAP
					tag = CHI
				}
			}
		}
	}

	option = {
		name = "We'll see how it plays out."
		ai_chance = {
			factor = 50
		}
	}
	option = {
		name = "The Germans are overstepping their boundaries."
		ai_chance = {
			factor = 50

				factor = 1.5
