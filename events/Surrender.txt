country_event = {
	
	id = 500
	
	trigger = {
		# Only allow going into exile if the faction leader is also at war with FROM!
		# Otherwise it could cause some trouble later on...
		or = {
			and = {
				faction = axis
				exists = GER
				GER = { faction = axis war_with = FROM }
			}
			and = {
				faction = allies
				or = {
					and = {
						exists = ENG
						ENG = { faction = allies war_with = FROM }
					}
					and = {
						exists = USA
						USA = { faction = allies war_with = FROM }
					}
				}
			}
			and = {
				faction = comintern
				exists = SOV
				SOV = { faction = comintern war_with = FROM }
			}
		}
		num_in_faction = 2
	}

	is_triggered_only = yes

	major = yes

	title = "EVTNAME500"
	desc = "EVTDESC500"
	picture = "greek_soldiers"

	option = {
		name = "EVTOPTB500"
		FROM = { country_event = 503 }
		form_government_in_exile = yes
	}	
}
